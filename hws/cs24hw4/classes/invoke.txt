call  8(%ebp) # This will call the getVolume function. It does this because
              # 8(%ebp) is the address of a shape object, and the first
              # member of shape objects are shape classes, and the first
              # member of shape classes is always the getVolume function
              # pointer. So the offset of the getVolume pointer in the shape
              # structs is 0, which means 8(%ebp) is the address of the
              # getVolume method, which means call 8(%ebp) invokes this method.
